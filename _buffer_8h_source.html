49  uint8_t* data = NULL) :

51  {
52  assert(t.width == 1 && "Can't create of a buffer of a vector type");
53  buf.elem_size = t.bits / 8;
54  size_t size = 1;
55  if (x_size) size *= x_size;
56  if (y_size) size *= y_size;
57  if (z_size) size *= z_size;
58  if (w_size) size *= w_size;
59  if (!data) {
60  allocation = (uint8_t *)calloc(buf.elem_size, size + 32/buf.elem_size);
61  buf.host = allocation;
62  while ((size_t)(buf.host) & 0x1f) buf.host++;
63  } else {
64  buf.host = data;
65  }
66  buf.dev = 0;
67  buf.host_dirty = false;
68  buf.dev_dirty = false;
69  buf.extent[0] = x_size;

71  buf.extent[2] = z_size;
72  buf.extent[3] = w_size;
73  buf.stride[0] = 1;
74  buf.stride[1] = x_size;
75  buf.stride[2] = x_size*y_size;
76  buf.stride[3] = x_size*y_size*z_size;
