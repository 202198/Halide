88 /** Coerce the two expressions to have the same type, using C-style
89  * casting rules. For the purposes of casting, a boolean type is
90  * UInt(1). We use the following procedure:
91  *
92  * If the types already match, do nothing.
93  *
94  * Then, if one type is a vector and the other is a scalar, the scalar
95  * is broadcast to match the vector width, and we continue.
96  *
97  * Then, if one type is floating-point and the other is not, the
98  * non-float is cast to the floating-point type, and we're done.
99  *
100  * Then, if neither is a float but one of the two is a constant, the
101  * constant is cast to match the non-const type and we're done. For
102  * example, e has type UInt(8), then (e*32) also has type UInt(8),
103  * despite the overflow that may occur. Note that this also means that
104  * (e*(-1)) is positive, and is equivalent to (e*255) - i.e. the (-1)
105  * is cast to a UInt(8) before the multiplication.
106  *
107  * Then, if both types are unsigned ints, the one with fewer bits is
108  * cast to match the one with more bits and we're done.
109  *
110  * Then, if both types are signed ints, the one with fewer bits is
111  * cast to match the one with more bits and we're done.
112  *
113  * Finally, if one type is an unsigned int and the other type is a signed
114  * int, both are cast to a signed int with the greater of the two
115  * bit-widths. For example, matching an Int(8) with a UInt(16) results
116  * in an Int(16).
117  *
118  */
119 void EXPORT match_types(Expr &a, Expr &b);
120 }

129 inline Expr cast(Type t, Expr a) {
130  assert(a.defined() && "cast of undefined");
131  if (a.type() == t) return a;

581  } else {

583  y = cast<float>(y);

585  }
