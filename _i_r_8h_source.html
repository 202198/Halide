855 /** A function call. This can represent a call to some extern
856  * function (like sin), but it's also our multi-dimensional
857  * version of a Load, so it can be a load from an input image, or
858  * a call to another halide function. The latter two types of call
859  * nodes don't survive all the way down to code generation - the
860  * lowering process converts them to Load nodes. */
861 struct Call : public ExprNode<Call> {
862  std::string name;
863  std::vector<Expr> args;
864  typedef enum {Image, Extern, Halide} CallType;
865  CallType call_type;
866 
867  // If it's a call to another halide function, this call node
868  // holds onto a pointer to that function
869  Function func;
870 
871  // If it's a call to an image, this call nodes hold a
872  // pointer to that image's buffer
873  Buffer image;
874 
875  // If it's a call to an image parameter, this call nodes holds a
876  // pointer to that
877  Parameter param;
878 
879  static Expr make(Type type, std::string name, const std::vector<Expr> &args, CallType call_type,
880  Function func, Buffer image, Parameter param) {
881  for (size_t i = 0; i < args.size(); i++) {
882  assert(args[i].defined() && "Call of undefined");
883  }
884  if (call_type == Halide) {
885  assert(func.value().defined() && "Call to undefined halide function");
886  assert(args.size() <= func.args().size() && "Call node with too many arguments.");
887  } else if (call_type == Image) {
888  assert((param.defined() || image.defined()) && "Call node to undefined image");
889  }
890 
891  Call *node = new Call;
892  node->type = type;
893  node->name = name;
894  node->args = args;
895  node->call_type = call_type;
896  node->func = func;
897  node->image = image;
898  node->param = param;
899  return node;
900  }
901 
902  /** Convenience constructor for calls to externally defined functions */
903  static Expr make(Type type, std::string name, const std::vector<Expr> &args) {
904  return make(type, name, args, Extern, Function(), Buffer(), Parameter());
905  }
906 
907  /** Convenience constructor for calls to other halide functions */
908  static Expr make(Function func, const std::vector<Expr> &args) {
909  assert(func.value().defined() && "Call to undefined halide function");
910  return make(func.value().type(), func.name(), args, Halide, func, Buffer(), Parameter());
911  }
912 
913  /** Convenience constructor for loads from concrete images */
914  static Expr make(Buffer image, const std::vector<Expr> &args) {
915  return make(image.type(), image.name(), args, Image, Function(), image, Parameter());
916  }
917 
918  /** Convenience constructor for loads from images parameters */
919  static Expr make(Parameter param, const std::vector<Expr> &args) {
920  return make(param.type(), param.name(), args, Image, Function(), Buffer(), param);
921  }
922 };
