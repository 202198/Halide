34  /** Prepare the buffer to be used as an image. Makes sure that the
35  * cached strides are correct, and that the image data is on the
36  * host. */
37  void prepare_for_direct_pixel_access() {
38  // Make sure buffer has been copied to host. This is a no-op
39  // if there's no device involved.
40  buffer.copy_to_host();
41 
42  // We're probably about to modify the pixels, so to be
43  // conservative we'd better set host dirty. If you're sure
44  // you're not going to modify this memory via the Image
45  // object, then you can call set_host_dirty(false) on the
46  // underlying buffer.
47  buffer.set_host_dirty(true);
48 
49  if (buffer.defined()) {

51  stride_1 = buffer.stride(1);
52  stride_2 = buffer.stride(2);
53  stride_3 = buffer.stride(3);
54  dims = buffer.dimensions();
55  } else {

57  stride_1 = stride_2 = stride_3 = 0;
58  dims = 0;
59  }
