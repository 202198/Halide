600  /** Compute this function as needed for each unique value of the
601  * given var for the given calling function f.
602  *
603  * For example, consider the simple pipeline:
604  \code
605  Func f, g;
606  Var x, y;
607  g(x, y) = x*y;
608  f(x, y) = g(x, y) + g(x, y+1) + g(x+1, y) + g(x+1, y+1);
609  \endcode
610  *
611  * If we schedule f like so:
612  *
613  \code
614  g.compute_at(f, x);
615  \endcode
616  *
617  * Then the C code equivalent to this pipeline will look like this
618  *
619  \code
620 
621  int f[height][width];
622  for (int y = 0; y < height; y++) {
623  for (int x = 0; x < width; x++) {
624  int g[2][2];
625  g[0][0] = x*y;
626  g[0][1] = (x+1)*y;
627  g[1][0] = x*(y+1);
628  g[1][1] = (x+1)*(y+1);
629  f[y][x] = g[0][0] + g[1][0] + g[0][1] + g[1][1];
630  }
631  }
632 
633  \endcode
634  *
635  * The allocation and computation of g is within f's loop over x,
636  * and enough of g is computed to satisfy all that f will need for
637  * that iteration. This has excellent locality - values of g are
638  * used as soon as they are computed, but it does redundant
639  * work. Each value of g ends up getting computed four times. If
640  * we instead schedule f like so:
641  *
642  \code
643  g.compute_at(f, y);
644  \endcode
645  *
646  * The equivalent C code is:
647  *
648  \code
649  int f[height][width];
650  for (int y = 0; y < height; y++) {
651  int g[2][width+1];
652  for (int x = 0; x < width; x++) {
653  g[0][x] = x*y;
654  g[1][x] = x*(y+1);
655  }
656  for (int x = 0; x < width; x++) {
657  f[y][x] = g[0][x] + g[1][x] + g[0][x+1] + g[1][x+1];
658  }
659  }
660  \endcode
661  *
662  * The allocation and computation of g is within f's loop over y,
663  * and enough of g is computed to satisfy all that f will need for
664  * that iteration. This does less redundant work (each point in g
665  * ends up being evaluated twice), but the locality is not quite
666  * as good, and we have to allocate more temporary memory to store
667  * g.
668  */
669  EXPORT Func &compute_at(Func f, Var var);

676  /** Compute all of this function once ahead of time. Reusing
677  * the example in \ref Func::compute_at :
678  *
679  \code
680  Func f, g;
681  Var x, y;
682  g(x, y) = x*y;
683  f(x, y) = g(x, y) + g(x, y+1) + g(x+1, y) + g(x+1, y+1);
684 
685  g.compute_root();
686  \endcode
687  *
688  * is equivalent to
689  *
690  \code
691  int f[height][width];
692  int g[height+1][width+1];
693  for (int y = 0; y < height+1; y++) {
694  for (int x = 0; x < width+1; x++) {
695  g[y][x] = x*y;
696  }
697  }
698  for (int y = 0; y < height; y++) {
699  for (int x = 0; x < width; x++) {
700  f[y][x] = g[y][x] + g[y+1][x] + g[y][x+1] + g[y+1][x+1];
701  }
702  }
703  \endcode
704  *
705  * g is computed once ahead of time, and enough is computed to
706  * satisfy all uses of it. This does no redundant work (each point
707  * in g is evaluated once), but has poor locality (values of g are
708  * probably not still in cache when they are used by f), and
709  * allocates lots of temporary memory to store g.
710  */
711  EXPORT Func &compute_root();

713  /** Allocate storage for this function within f's loop over
714  * var. Scheduling storage is optional, and can be used to
715  * separate the loop level at which storage occurs from the loop
716  * level at which computation occurs to trade off between locality
717  * and redundant work. This can open the door for two types of
718  * optimization.
719  *
720  * Consider again the pipeline from \ref Func::compute_at :
721  \code
722  Func f, g;
723  Var x, y;
724  g(x, y) = x*y;
725  f(x, y) = g(x, y) + g(x+1, y) + g(x, y+1) + g(x+1, y+1);
726  \endcode
727  *
728  * If we schedule it like so:
729  *
730  \code
731  g.compute_at(f, x).store_at(f, y);
732  \endcode
733  *
734  * Then the computation of g takes place within the loop over x,
735  * but the storage takes place within the loop over y:
736  *
737  \code
738  int f[height][width];
739  for (int y = 0; y < height; y++) {
740  int g[2][width+1];
741  for (int x = 0; x < width; x++) {
742  g[0][x] = x*y;
743  g[0][x+1] = (x+1)*y;
744  g[1][x] = x*(y+1);
745  g[1][x+1] = (x+1)*(y+1);
746  f[y][x] = g[0][x] + g[1][x] + g[0][x+1] + g[1][x+1];
747  }
748  }
749  \endcode
750  *
751  * Provided the for loop over x is serial, halide then
752  * automatically performs the following sliding window
753  * optimization:
754  *
755  \code
756  int f[height][width];
757  for (int y = 0; y < height; y++) {
758  int g[2][width+1];
759  for (int x = 0; x < width; x++) {
760  if (x == 0) {
761  g[0][x] = x*y;
762  g[1][x] = x*(y+1);
763  }
764  g[0][x+1] = (x+1)*y;
765  g[1][x+1] = (x+1)*(y+1);
766  f[y][x] = g[0][x] + g[1][x] + g[0][x+1] + g[1][x+1];
767  }
768  }
769  \endcode
770  *
771  * Two of the assignments to g only need to be done when x is
772  * zero. The rest of the time, those sites have already been
773  * filled in by a previous iteration. This version has the
774  * locality of compute_at(f, x), but allocates more memory and
775  * does much less redundant work.
776  *
777  * Halide then further optimizes this pipeline like so:
778  *
779  \code
780  int f[height][width];
781  for (int y = 0; y < height; y++) {
782  int g[2][2];
783  for (int x = 0; x < width; x++) {
784  if (x == 0) {
785  g[0][0] = x*y;
786  g[1][0] = x*(y+1);
787  }
788  g[0][(x+1)%2] = (x+1)*y;
789  g[1][(x+1)%2] = (x+1)*(y+1);
790  f[y][x] = g[0][x%2] + g[1][x%2] + g[0][(x+1)%2] + g[1][(x+1)%2];
791  }
792  }
793  \endcode
794  *
795  * Halide has detected that it's possible to use a circular buffer
796  * to represent g, and has reduced all accesses to g modulo 2 in
797  * the x dimension. This optimization only triggers if the for
798  * loop over x is serial, and if halide can statically determine
799  * some power of two large enough to cover the range needed. For
800  * powers of two, the modulo operator compiles to more efficient
801  * bit-masking. This optimization reduces memory usage, and also
802  * improves locality by reusing recently-accessed memory instead
803  * of pulling new memory into cache.
804  *
805  */
806  EXPORT Func &store_at(Func f, Var var);

809  * schedules storage within the loop over a dimension of a
810  * reduction domain */
811  EXPORT Func &store_at(Func f, RVar var);

821  *
822  * Consider once more the pipeline from \ref Func::compute_at :

824  \code
